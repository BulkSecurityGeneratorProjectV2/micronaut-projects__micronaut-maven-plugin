11  import java.util.HashMap;
12  import java.util.Map;
13  import java.util.Optional;
14  
15  /**
16   * Parses the Micronaut application configuration.
17   *
18   * @author Álvaro Sánchez-Mariscal
19   * @since 1.1
20   */
21  @Singleton
22  public class ApplicationConfigurationService {
23  
24      public static final String DEFAULT_PORT = "8080";
25  
26      private final MavenProject mavenProject;
27      private final Map<String, Object> applicationConfiguration;
28  
29      @SuppressWarnings("CdiInjectionPointsInspection")
30      @Inject
31      public ApplicationConfigurationService(MavenProject mavenProject) {
32          this.mavenProject = mavenProject;
33          this.applicationConfiguration = parseApplicationConfiguration();
34      }
35  
36      public String getServerPort() {
37          return applicationConfiguration.getOrDefault("MICRONAUT_SERVER_PORT", applicationConfiguration.getOrDefault("micronaut.server.port", DEFAULT_PORT)).toString();
38      }
39  
40      private Map<String, Object> parseApplicationConfiguration() {
41          Map<String, Object> configuration = new HashMap<>();
42  
43          PropertySourceLoader[] loaders = {
44                  new YamlPropertySourceLoader(),
45                  new PropertiesPropertySourceLoader()
46          };
47  
48          for (PropertySourceLoader loader : loaders) {
49              Optional<PropertySource> propertySource = loader.load("application", new DefaultFileSystemResourceLoader(new File(mavenProject.getBasedir(), "src/main/resources")));
50              if (propertySource.isPresent()) {
51                  MapPropertySource mapPropertySource = (MapPropertySource) propertySource.get();
52                  configuration.putAll(mapPropertySource.asMap());
53              }
54          }
55  
56          MapPropertySource[] propertySources = {
57                  new EnvironmentPropertySource(),
58                  new SystemPropertiesPropertySource()
59          };
60          for (MapPropertySource propertySource : propertySources) {
61              configuration.putAll(propertySource.asMap());
62          }
63  
64          return configuration;
65      }
66  }
