17  /**
18   * <p>Executes a Micronaut application in development mode.</p>
19   *
20   * <p>It watches for changes in the project tree. If there are changes in the {@code pom.xml} file, dependencies will be reloaded. If
21   * the changes are anywhere underneath {@code src/main}, it will recompile the project and restart the application.</p>
22   *
23   * <p>The plugin can handle changes in all the languages supported by Micronaut: Java, Kotlin and Groovy.</p>
24   *
25   * @author Álvaro Sánchez-Mariscal
26   * @since 1.0.0
27   */
28  @SuppressWarnings("unused")
29  @Mojo(name = "run", requiresDependencyResolution = ResolutionScope.COMPILE_PLUS_RUNTIME, defaultPhase = LifecyclePhase.PREPARE_PACKAGE)
30  public class RunMojo extends AbstractRunMojo {
31  
32      @Inject
33      public RunMojo(MavenProject mavenProject, MavenSession mavenSession, BuildPluginManager pluginManager, ProjectDependenciesResolver resolver, ProjectBuilder projectBuilder, ToolchainManager toolchainManager, CompilerService compilerService, ExecutorService executorService) {
34          super(mavenProject, mavenSession, pluginManager, resolver, projectBuilder, toolchainManager, compilerService, executorService);
35          addClasses = true;
36      }
37  }
