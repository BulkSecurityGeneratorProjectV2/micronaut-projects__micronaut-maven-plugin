32      @Override
33      public ContainerBuildPlan extendContainerBuildPlan(ContainerBuildPlan buildPlan, Map<String, String> properties,
34                                                         Optional<Void> extraConfig, MavenData mavenData,
35                                                         ExtensionLogger logger) throws JibPluginExtensionException {
36  
37          ContainerBuildPlan.Builder builder = buildPlan.toBuilder();
38          MicronautRuntime runtime = MicronautRuntime.valueOf(mavenData.getMavenProject().getProperties().getProperty(MicronautRuntime.PROPERTY, "none").toUpperCase());
39  
40          JibConfigurationServicenfigurationService">JibConfigurationService jibConfigurationService = new JibConfigurationService(mavenData.getMavenProject());
41          String from = jibConfigurationService.getFromImage().orElse(DEFAULT_BASE_IMAGE);
42  
43          ApplicationConfigurationServicenfigurationService">ApplicationConfigurationService applicationConfigurationService = new ApplicationConfigurationService(mavenData.getMavenProject());
44          Map<String, Object> applicationConfiguration = applicationConfigurationService.getApplicationConfiguration();
45  
46          int port = Integer.parseInt(applicationConfiguration.getOrDefault("MICRONAUT_SERVER_PORT", applicationConfiguration.getOrDefault("micronaut.server.port", 8080)).toString());
47          logger.log(ExtensionLogger.LogLevel.LIFECYCLE, "Exposing port: " + port);
48          builder.setBaseImage(from);
49          if (port > 0) {
50              builder.addExposedPort(Port.tcp(port));
51          }
52  
53          switch (runtime.getBuildStrategy()) {
54              case ORACLE_FUNCTION:
55  
56                  List<? extends LayerObject> originalLayers = buildPlan.getLayers();
57                  builder.setLayers(Collections.emptyList());
58  
59                  for (LayerObject layer : originalLayers) {
60                      builder.addLayer(remapLayer(layer));
61                  }
62  
63                  builder.setBaseImage("fnproject/fn-java-fdk:" + determineProjectFnVersion())
64                          .setWorkingDirectory(AbsoluteUnixPath.get("/function"))
65                          .setEntrypoint(null)
66                          .setCmd(Collections.singletonList("io.micronaut.oraclecloud.function.http.HttpFunction::handleRequest"));
67                  break;
68              case LAMBDA:
69                  List<String> entrypoint = buildPlan.getEntrypoint();
70                  Objects.requireNonNull(entrypoint).set(entrypoint.size() - 1, "io.micronaut.function.aws.runtime.MicronautLambdaRuntime");
71                  builder.setEntrypoint(entrypoint);
72                  break;
73          }
74          return builder.build();
75      }
76  
77      private LayerObject remapLayer(LayerObject layerObject) {
78          FileEntriesLayer originalLayer = (FileEntriesLayer) layerObject;
79          FileEntriesLayer.Builder builder = FileEntriesLayer.builder().setName(originalLayer.getName());
80          for (FileEntry originalEntry : originalLayer.getEntries()) {
81              builder.addEntry(remapEntry(originalEntry));
82          }
83  
84          return builder.build();
85      }
86  
87      private FileEntry remapEntry(FileEntry originalEntry) {
88          List<String> pathComponents = UnixPathParser.parse(originalEntry.getExtractionPath().toString());
89  
90          AbsoluteUnixPath newPath = AbsoluteUnixPath.get("/function/app/" + pathComponents.get(pathComponents.size() - 1));
91          return new FileEntry(originalEntry.getSourceFile(), newPath, originalEntry.getPermissions(),
92                  originalEntry.getModificationTime(), originalEntry.getOwnership());
93      }
94  
95      private String determineProjectFnVersion() {
96          ArtifactVersion javaVersion = new DefaultArtifactVersion(System.getProperty("java.version"));
97          if (javaVersion.getMajorVersion() >= 11) {
98              return "jre11-latest";
99          } else {
100             return "latest";
101         }
102     }
