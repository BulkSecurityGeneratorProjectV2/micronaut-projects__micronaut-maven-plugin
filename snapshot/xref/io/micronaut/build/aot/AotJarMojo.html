41  /**
42   * Generates a jar optimized by Micronaut AOT, which is a jar corresponding to the regular application jar, except that
43   * it contains some optimizations computed at build time. It may contain, for example, additional classes, or even have
44   * different resources.
45   */
46  @Mojo(name = "aot-jar", defaultPhase = LifecyclePhase.PACKAGE, requiresDependencyResolution = ResolutionScope.COMPILE_PLUS_RUNTIME)
47  public class AotJarMojo extends AbstractMicronautAotMojo {
48  
49      /**
50       * The filename (excluding the extension, and with no path information) that the produced artifact will be called.
51       */
52      @Parameter( defaultValue = "${project.build.finalName}", readonly = true )
53      private String finalName;
54  
55      /**
56       * The Jar archiver.
57       */
58      @Component
59      private Map<String, Archiver> archivers;
60  
61      private final ExecutorService executorService;
62  
63      @Inject
64      @SuppressWarnings("CdiInjectionPointsInspection")
65      public AotJarMojo(CompilerService compilerService, ExecutorService executorService, MavenProject mavenProject,
66                        MavenSession mavenSession, RepositorySystem repositorySystem) {
67          super(compilerService, mavenProject, mavenSession, repositorySystem);
68          this.executorService = executorService;
69      }
70  
71      @Override
72      protected void doExecute() throws DependencyResolutionException, MojoExecutionException {
73          executorService.executeGoal("io.micronaut.build:micronaut-maven-plugin", "aot-analysis");
74          MavenArchiver archiver = new MavenArchiver();;
75          try {
76              configureAotJar(archiver);
77              archiver.createArchive(mavenSession, mavenProject, new MavenArchiveConfiguration());
78          } catch (ManifestException | IOException | DependencyResolutionRequiredException e) {
79              throw new MojoExecutionException("Unable to package AOT jar", e);
80          }
81      }
82  
83      private void configureAotJar(MavenArchiver archiver) throws MojoExecutionException {
84          Archiver jarArchiver = archivers.get("jar");
85          jarArchiver.addFileSet(new DefaultFileSet(outputFile("generated/classes")));
86          File unmodifiedJar = mavenProject.getArtifact().getFile();
87          DefaultArchivedFileSet archivedFileSet = new DefaultArchivedFileSet(unmodifiedJar);
88          configureExcludes(archivedFileSet);
89          jarArchiver.addArchivedFileSet(archivedFileSet);
90          archiver.setArchiver((JarArchiver) jarArchiver);
91          File aotJar = new File(unmodifiedJar.getParentFile(), finalName + "-aot.jar");
92          archiver.setOutputFile(aotJar);
93      }
94  
95      /**
96       * This method reads the resource-filter.txt file which is generated by the AOT analysis
97       * and configures the archive to exclude the resources listed in that file
98       * @param archivedFileSet the archive configuration
99       * @throws MojoExecutionException in case the resource filter cannot be read
100      */
101     private void configureExcludes(DefaultArchivedFileSet archivedFileSet) throws MojoExecutionException {
102         File excluded = outputFile("generated/logs/resource-filter.txt");
103         if (excluded.exists()) {
104             Set<String> excludes = new HashSet<>();
105             try (BufferedReader reader = new BufferedReader(new InputStreamReader(new FileInputStream(excluded)))) {
106                 String line;
107                 while ((line = reader.readLine()) != null) {
108                     line = line.trim();
109                     if (!line.isEmpty()) {
110                         excludes.add(line);
111                     }
112                 }
113             } catch (IOException e) {
114                 throw new MojoExecutionException("Unable to read filter file", e);
115             }
116             if (!excludes.isEmpty()) {
117                 archivedFileSet.setExcludes(excludes.toArray(new String[0]));
118             }
119         }
120     }
