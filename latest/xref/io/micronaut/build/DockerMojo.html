14  import javax.inject.Inject;

18  /**
19   * <p>Allows using a provided Dockerfile.</p>
20   * <p><strong>WARNING</strong>: this goal is not intended to be executed directly. Instead, specify the packaging type
21   * using the <code>packaging</code> property, eg:</p>
22   *
23   * <pre>mvn package -Dpackaging=docker</pre>
24   *
25   * @author Álvaro Sánchez-Mariscal
26   * @since 1.1
27   */
28  @Mojo(name = DockerMojo.DOCKER_PACKAGING, requiresDependencyResolution = ResolutionScope.COMPILE_PLUS_RUNTIME)
29  public class DockerMojo extends AbstractDockerMojo {
30  
31      public static final String DOCKER_PACKAGING = "docker";
32  
33      @SuppressWarnings("CdiInjectionPointsInspection")
34      @Inject
35      public DockerMojo(MavenProject mavenProject, JibConfigurationService jibConfigurationService,
36                        ApplicationConfigurationService applicationConfigurationService, DockerService dockerService) {
37          super(mavenProject, jibConfigurationService, applicationConfigurationService, dockerService);
38      }
39  
40      @Override
41      public void execute() throws MojoExecutionException {
42          File dockerfile = new File(mavenProject.getBasedir(), DockerfileMojo.DOCKERFILE);
43          if (dockerfile.exists()) {
44              try {
45                  getLog().info("Using provided Dockerfile: " + dockerfile.getAbsolutePath());
46                  mavenProject.getProperties().put(PropertyNames.SKIP, "true");
47  
48                  copyDependencies();
49  
50                  String targetDir = mavenProject.getBuild().getDirectory();
51                  File targetDockerfile = new File(targetDir, dockerfile.getName());
52                  FileUtils.copyFile(dockerfile, targetDockerfile);
53  
54                  BuildImageCmd buildImageCmd = dockerService.buildImageCmd()
55                          .withDockerfile(targetDockerfile)
56                          .withTags(getTags())
57                          .withBaseDirectory(new File(targetDir));
58                  dockerService.buildImage(buildImageCmd);
59              } catch (IOException e) {
60                  throw new MojoExecutionException(e.getMessage(), e);
61              }
62          }
63      }
64  
65  }
